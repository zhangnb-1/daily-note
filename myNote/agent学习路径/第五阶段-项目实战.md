#### 5.1.2.2 Agent基类

```python
# src/agents/base.py
from abc import ABC, abstractmethod
from typing import Optional, Dict, Any
from langchain_openai import ChatOpenAI
from langchain_core.messages import HumanMessage, SystemMessage
from langchain.memory import ConversationBufferMemory

class BaseAgent(ABC):
    """Agent基类"""

    def __init__(self, name: str, description: str, llm: Optional[ChatOpenAI] = None):
        self.name = name
        self.description = description
        self.llm = llm or ChatOpenAI(temperature=0)
        self.memory = ConversationBufferMemory(
            return_messages=True,
            output_key="response",
            input_key="input"
        )

    @abstractmethod
    def get_system_prompt(self) -> str:
        """获取系统提示词"""
        pass

    @abstractmethod
    def process(self, input_text: str, context: Optional[Dict] = None) -> str:
        """处理输入"""
        pass

    def think(self, task: str, tools: list = None) -> str:
        """使用ReAct模式思考"""
        prompt = f"""
        你是一个{self.name}，{self.description}

        当前任务: {task}

        请按照以下步骤思考：
        1. 分析任务需求
        2. 确定需要的工具
        3. 执行操作
        4. 返回结果
        """

        if tools:
            prompt += f"\n可用工具: {', '.join([t.name for t in tools])}"

        response = self.llm.invoke([
            SystemMessage(content=self.get_system_prompt()),
            HumanMessage(content=prompt)
        ])

        return response.content

    def add_to_memory(self, input_text: str, response: str):
        """添加到记忆"""
        self.memory.save_context(
            {"input": input_text},
            {"response": response}
        )

    def get_history(self) -> list:
        """获取对话历史"""
        return self.memory.load_memory_variables({}).get("history", [])
```

#### 5.1.2.3 日历Agent

```python
# src/agents/calendar.py
from datetime import datetime, timedelta
from typing import Optional, List
from .base import BaseAgent

class CalendarEvent:
    def __init__(self, title: str, start_time: datetime, end_time: datetime, description: str = ""):
        self.title = title
        self.start_time = start_time
        self.end_time = end_time
        self.description = description

class CalendarAgent(BaseAgent):
    """日历Agent - 管理日程安排"""

    def __init__(self, llm=None):
        super().__init__(
            name="日历助手",
            description="帮助用户管理日程安排"
        )
        self.events: List[CalendarEvent] = []

    def get_system_prompt(self) -> str:
        return """你是一个专业的日历助手，帮助用户：
        1. 创建和查看日程事件
        2. 设置会议和提醒
        3. 检查时间冲突
        4. 提供日程建议

        始终用友好、专业的语气与用户交流。"""

    def add_event(self, title: str, start_time: datetime, end_time: datetime, description: str = "") -> str:
        """添加事件"""
        event = CalendarEvent(title, start_time, end_time, description)
        self.events.append(event)
        return f"已添加日程：{title}，时间：{start_time.strftime('%Y-%m-%d %H:%M')} - {end_time.strftime('%H:%M')}"

    def get_events(self, date: Optional[datetime] = None) -> List[CalendarEvent]:
        """获取指定日期的事件"""
        if date is None:
            return self.events

        return [
            e for e in self.events
            if e.start_time.date() == date.date()
        ]

    def check_conflict(self, start_time: datetime, end_time: datetime) -> List[CalendarEvent]:
        """检查时间冲突"""
        conflicts = []
        for event in self.events:
            if (start_time < event.end_time and end_time > event.start_time):
                conflicts.append(event)
        return conflicts

    def process(self, input_text: str, context: Optional[dict] = None) -> str:
        """处理用户请求"""
        # 简化处理 - 实际项目中可以结合LLM解析意图
        if "今天" in input_text or "today" in input_text.lower():
            today_events = self.get_events(datetime.now())
            if not today_events:
                return "今天没有日程安排"
            return "\n".join([
                f"- {e.title}: {e.start_time.strftime('%H:%M')} - {e.end_time.strftime('%H:%M')}"
                for e in today_events
            ])

        return self.think(input_text)

    def suggest_time(self, duration: int, preferred_time: str = "上午") -> str:
        """建议可用时间"""
        now = datetime.now()
        suggested = now.replace(hour=10 if preferred_time == "上午" else 14)

        conflicts = self.check_conflict(suggested, suggested + timedelta(minutes=duration))

        if conflicts:
            return f"{preferred_time}{duration}分钟内暂无空闲时间"

        return f"建议时间：{suggested.strftime('%Y-%m-%d %H:%M')}"
```

```python
# src/agents/email.py
from dataclasses import dataclass
from typing import Optional, List
from datetime import datetime
from .base import BaseAgent

@dataclass
class Email:
    id: str
    sender: str
    subject: str
    body: str
    timestamp: datetime
    read: bool = False
    starred: bool = False

class EmailAgent(BaseAgent):
    """邮件Agent"""

    def __init__(self, llm=None):
        super().__init__(
            name="邮件助手",
            description="帮助用户管理和发送邮件"
        )
        self.emails: List[Email] = []
        self.draft: Optional[Email] = None

    def get_system_prompt(self) -> str:
        return """你是一个专业的邮件助手，帮助用户：
        1. 阅读和搜索邮件
        2. 撰写和发送邮件
        3. 管理邮件夹
        4. 标记重要邮件

        注意保护用户隐私，不要读取邮件正文内容。"""

    def add_email(self, sender: str, subject: str, body: str) -> str:
        """模拟接收邮件"""
        email = Email(
            id=f"email_{len(self.emails) + 1}",
            sender=sender,
            subject=subject,
            body=body,
            timestamp=datetime.now()
        )
        self.emails.append(email)
        return f"收到新邮件 from {sender}: {subject}"

    def get_unread(self) -> List[Email]:
        """获取未读邮件"""
        return [e for e in self.emails if not e.read]

    def search(self, keyword: str) -> List[Email]:
        """搜索邮件"""
        results = []
        for email in self.emails:
            if keyword.lower() in email.subject.lower() or keyword.lower() in email.body.lower():
                results.append(email)
        return results

    def compose(self, to: str, subject: str, body: str) -> str:
        """撰写邮件"""
        self.draft = Email(
            id="draft",
            sender="me",
            subject=subject,
            body=body,
            timestamp=datetime.now()
        )
        return f"邮件草稿已创建：\n收件人：{to}\n主题：{subject}"

    def send(self) -> str:
        """发送邮件"""
        if not self.draft:
            return "没有待发送的邮件"

        self.emails.insert(0, self.draft)
        sent = self.draft
        self.draft = None
        return f"邮件已发送：{sent.subject}"

    def process(self, input_text: str, context: Optional[dict] = None) -> str:
        """处理用户请求"""
        input_lower = input_text.lower()

        if "未读" in input_text:
            unread = self.get_unread()
            if not unread:
                return "没有未读邮件"
            return "\n".join([
                f"📧 {e.sender}: {e.subject}"
                for e in unread[:5]
            ])

        if "搜索" in input_text or "search" in input_lower:
            keyword = input_text.replace("搜索", "").replace("search", "").strip()
            results = self.search(keyword)
            if not results:
                return f"没有找到关于'{keyword}'的邮件"
            return "\n".join([
                f"📧 {e.sender}: {e.subject}"
                for e in results[:5]
            ])

        return self.think(input_text)
```

#### 5.1.2.5 任务路由器

```python
# src/router/task_router.py
from typing import Dict, Optional
from langchain_openai import ChatOpenAI
from langchain_core.messages import HumanMessage, SystemMessage
from langchain_core.output_parsers import StrOutputParser

class TaskRouter:
    """任务路由器 - 将用户请求路由到合适的Agent"""

    def __init__(self, agents: Dict):
        self.agents = agents
        self.llm = ChatOpenAI(temperature=0)

        # 定义路由规则
        self.route_rules = {
            "日历": ["日程", "会议", " calendar", "schedule", "安排", "时间"],
            "邮件": ["邮件", " email", "写信", "发送", "收件"],
            "搜索": ["搜索", " search", "查找", "找一下", "google"],
            "提醒": ["提醒", " remember", "通知", "闹铃"]
        }

    def route(self, user_input: str) -> str:
        """路由到合适的Agent"""
        # 1. 关键词匹配
        for agent_name, keywords in self.route_rules.items():
            for keyword in keywords:
                if keyword.lower() in user_input.lower():
                    return agent_name

        # 2. LLM智能路由
        return self._llm_route(user_input)

    def _llm_route(self, user_input: str) -> str:
        """使用LLM进行路由"""
        prompt = f"""
        根据用户输入，判断应该使用哪个Agent处理：

        用户输入：{user_input}

        可用Agent：
        - 日历：处理日程、会议、时间相关
        - 邮件：处理邮件收发
        - 搜索：处理信息搜索
        - 提醒：处理提醒设置

        只返回一个Agent名称，不要其他内容。
        """

        response = self.llm.invoke([
            SystemMessage(content="你是一个任务路由助手"),
            HumanMessage(content=prompt)
        ])

        result = response.content.strip()
        return result if result in self.agents else "搜索"

    def process(self, user_input: str) -> str:
        """处理用户请求"""
        # 路由
        agent_name = self.route(user_input)
        agent = self.agents.get(agent_name)

        if not agent:
            return f"无法处理请求：找不到{agent_name}Agent"

        # 处理
        return agent.process(user_input)
```

## 5.2 项目二：RAG知识库（15-20小时）

### 5.2.1 项目概述

企业内部知识问答系统，支持文档上传、智能问答。

### 5.2.2 核心代码

```python
# src/rag_knowledge_base.py
from langchain_openai import OpenAIEmbeddings, ChatOpenAI
from langchain_community.vectorstores import Chroma
from langchain_community.document_loaders import PyPDFLoader, TextLoader
from langchain.text_splitter import RecursiveCharacterTextSplitter
from langchain_core.prompts import ChatPromptTemplate
from langchain_core.output_parsers import StrOutputParser
from langchain.chains import RetrievalQA

class KnowledgeBase:
    """RAG知识库系统"""

    def __init__(self, persist_directory="./chroma_db"):
        self.embeddings = OpenAIEmbeddings(model="text-embedding-3-small")
        self.llm = ChatOpenAI(model="gpt-4", temperature=0)
        self.persist_directory = persist_directory
        self.vectorstore = None
        self.splitter = RecursiveCharacterTextSplitter(
            chunk_size=500,
            chunk_overlap=50,
            separators=["\n\n", "\n", "。", "，"]
        )

    def load_document(self, filepath: str):
        """加载文档"""
        if filepath.endswith(".pdf"):
            loader = PyPDFLoader(filepath)
        else:
            loader = TextLoader(filepath, encoding="utf-8")

        documents = loader.load()
        chunks = self.splitter.split_documents(documents)

        # 存储到向量数据库
        self.vectorstore = Chroma.from_documents(
            documents=chunks,
            embedding=self.embeddings,
            persist_directory=self.persist_directory
        )
        self.vectorstore.persist()

        return f"已加载 {len(chunks)} 个文档块"

    def query(self, question: str) -> str:
        """问答"""
        if not self.vectorstore:
            return "请先加载文档"

        # 构建检索链
        retriever = self.vectorstore.as_retriever(search_kwargs={"k": 3})

        # RAG Prompt
        prompt = ChatPromptTemplate.from_messages([
            ("system", """你是一个专业的知识库助手。
            基于以下参考材料回答用户的问题。
            如果无法从材料中找到答案，请如实说明。

            参考材料：
            {context}"""),
            ("human", "{question}")
        ])

        # 组装链
        def format_docs(docs):
            return "\n\n".join([
                f"材料{i+1}: {doc.page_content}"
                for i, doc in enumerate(docs)
            ])

        chain = (
            {"context": retriever | format_docs, "question": lambda x: x}
            | prompt
            | self.llm
            | StrOutputParser()
        )

        return chain.invoke(question)

# 使用
kb = KnowledgeBase()
kb.load_document("./documents/公司制度.pdf")
answer = kb.query("公司的年假政策是什么？")
print(answer)
```

## 5.3 项目三：Multi-Agent调度（20-30小时）

### 5.3.1 项目概述

多Agent协作完成复杂任务，如市场调研报告生成。

### 5.3.2 核心代码

```python
# src/multi_agent_system.py
from crewai import Agent, Task, Crew, Process
from langchain_openai import ChatOpenAI
from dataclasses import dataclass
from typing import List

@dataclass
class TaskResult:
    task_name: str
    agent_name: str
    result: str
    status: str

class MultiAgentSystem:
    """Multi-Agent调度系统"""

    def __init__(self):
        self.llm = ChatOpenAI(model="gpt-4")

    def create_research_crew(self, topic: str) -> Crew:
        """创建市场调研团队"""
        # 定义Agent
        researcher = Agent(
            role="高级研究员",
            goal=f"深入研究{topic}领域",
            backstory="你是一位资深的行业研究员，擅长收集和分析信息",
            llm=self.llm,
            verbose=True
        )

        analyst = Agent(
            role="数据分析师",
            goal="从数据中发现洞察",
            backstory="你擅长数据分析，能从数据中提取有价值的洞察",
            llm=self.llm,
            verbose=True
        )

        writer = Agent(
            role="报告撰写专家",
            goal="撰写专业的市场调研报告",
            backstory="你是一位专业的商业写作专家",
            llm=self.llm,
            verbose=True
        )

        # 定义任务
        task1 = Task(
            description=f"收集关于{topic}的全面信息，包括市场规模、竞争格局、发展趋势",
            agent=researcher,
            expected_output="详细的研究报告"
        )

        task2 = Task(
            description="分析研究报告，提取5个关键洞察",
            agent=analyst,
            expected_output="关键洞察列表"
        )

        task3 = Task(
            description="基于洞察撰写完整的{topic}市场调研报告",
            agent=writer,
            expected_output="完整报告"
        )

        # 创建团队
        crew = Crew(
            agents=[researcher, analyst, writer],
            tasks=[task1, task2, task3],
            process=Process.hierarchical
        )

        return crew

    def run(self, topic: str) -> str:
        """执行调研"""
        crew = self.create_research_crew(topic)
        result = crew.kickoff()
        return result

# 使用
system = MultiAgentSystem()
report = system.run("AI Agent市场")
print(report)
```

### 5.4.1 项目选择建议

| 项目 | 难度 | 建议 | 收获 |

| 个人助手Agent | ⭐⭐ | 必做 | 掌握Agent核心架构 |

### 5.4.2 常见问题与解决

| 问题 | 原因 | 解决方案 |
|------|------|----------|
| Agent响应慢 | LLM调用延迟 | 添加流式输出、优化Prompt |
| 工具调用失败 | API限流或错误 | 添加重试机制、错误处理 |
| 记忆丢失 | 内存未持久化 | 使用向量数据库持久化 |
| 多Agent冲突 | 任务分配不明确 | 明确Agent角色和边界 |
